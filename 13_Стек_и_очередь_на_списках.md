        }

    private Node head;
    private Node tail;

    private static class Node {
        int value;
        Node next;

        Node(int value) {
            this.value = value;
        }
    }

    public void enqueue(int value) {
        Node newNode = new Node(value);
        if (tail != null) {
            tail.next = newNode;
        } else {
            head = newNode;
        }
        tail = newNode;
    }

    public int dequeue() {
        if (head == null) throw new NoSuchElementException();
        int val = head.value;
        head = head.next;
        if (head == null) tail = null;
        return val;
    }

    public int peek() {
        if (head == null) throw new NoSuchElementException();
        return head.value;
    }
}
